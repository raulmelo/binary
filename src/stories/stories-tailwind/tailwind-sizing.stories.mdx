# Width
Utilities for setting the width of an element

## Usage
Control the font size of an element using the `width-{name}` utilities.

```html
  <div class="w-screen"></div>
```

export const TemplateWidth = () => {
  const values = [{"class":"88","value":"88px"},{"class":"100","value":"100px"},{"class":"auto","value":"auto"},{"class":"px","value":"1px"},{"class":"xs","value":"8px"},{"class":"sm","value":"16px"},{"class":"md","value":"24px"},{"class":"lg","value":"32px"},{"class":"xl","value":"40px"},{"class":"1/2","value":"50%"},{"class":"1/3","value":"33.333333%"},{"class":"2/3","value":"66.666667%"},{"class":"1/4","value":"25%"},{"class":"2/4","value":"50%"},{"class":"3/4","value":"75%"},{"class":"1/5","value":"20%"},{"class":"2/5","value":"40%"},{"class":"3/5","value":"60%"},{"class":"4/5","value":"80%"},{"class":"1/6","value":"16.666667%"},{"class":"2/6","value":"33.333333%"},{"class":"3/6","value":"50%"},{"class":"4/6","value":"66.666667%"},{"class":"5/6","value":"83.333333%"},{"class":"1/12","value":"8.333333%"},{"class":"2/12","value":"16.666667%"},{"class":"3/12","value":"25%"},{"class":"4/12","value":"33.333333%"},{"class":"5/12","value":"41.666667%"},{"class":"6/12","value":"50%"},{"class":"7/12","value":"58.333333%"},{"class":"8/12","value":"66.666667%"},{"class":"9/12","value":"75%"},{"class":"10/12","value":"83.333333%"},{"class":"11/12","value":"91.666667%"},{"class":"full","value":"100%"},{"class":"screen","value":"100vw"}]
  return <div>
    {values.map( (item, index) => {
      return (
      <div key={index}>
        <div class="tailwindStorybook__sizing">
          <div class="tailwindStorybook__sizing--h">
            <div class={`tailwindStorybook__sizing--label w-${item.class}`}></div>
            <span>
              <input onClick={ () => copyClass(`w-${item.class}`) } readOnly={true} id={`w-${item.class}`} defaultValue={`w-${item.class}`} />
              <small> value: {item.value}</small>
            </span>
          </div>
        </div>
      </div>
      )
    })}
  </div>
}

<Story name="Width">
  {TemplateWidth.bind()}
</Story>
